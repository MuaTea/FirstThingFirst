#include <stdio.h>
#include <string.h>
#define MAX_LEN 100
#include <stdlib.h>
#include <time.h>

int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int main() {
    int arr[MAX_LEN];
    int *ptr=arr;
    int min=1;
    int max=100;
    srand(time((NULL)));
    for (int i=0;i<MAX_LEN;i++) {
        arr[i]=rand()%(max-min+1);
    }
    int n = sizeof(arr) / sizeof(arr[0]);
    qsort(arr, n, sizeof(int), compare);

    for (int i=0;i<n;i++) {
        printf("%d ",*(ptr+i));
    }
    printf("\n");

    return 0;
}
