#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define MAX_LEN 100

int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int main() {
    int arr[MAX_LEN];
    srand(time(NULL));  // 初始化随机数种子

    // 填充数组
    for (int i = 0; i < MAX_LEN; i++) {
        arr[i] = rand() % 100 + 1;  // 生成1到100之间的随机数
    }

    int n = sizeof(arr) / sizeof(arr[0]);

    // 排序数组
    qsort(arr, n, sizeof(int), compare);

    // 打印排序后的数组
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}
