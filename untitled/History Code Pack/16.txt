void Sort(int arr[][3],int rows) {
    int* a =Assignment(rows);
    int res=0;
    //int* temp=&res;
    for (int i=0;i<rows;i++) {
        for(int j=0;j<3;j++) {
            res += *a;
        }
    }
    //return temp;
    //printf("%d",*temp);
}

int Ran(const int MIN,const int MAX) {
    srand(time(NULL));
    int ran=rand()%(MAX-MIN+1);
    return ran;
}

int* Assignment(int rows) {
    int* temp=arr;
    for (int i=0;i<rows;i++) {
        for(int j=0;j<3;j++) {
            arr[i][j]=Ran(1,100);
        }
    }
    return temp;
}
