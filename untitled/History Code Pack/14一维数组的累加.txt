#include <stdio.h>
#include <string.h>
#define MAX_LEN 100
#include <ctype.h>
#include <stdlib.h>
#include <time.h>

void Cal(int *arr);
void Ran();
int arr[MAX_LEN];
int res=0;

int main(/*int argc,char* argv[]*/) {
    Ran();
    Cal(arr);
    return 0;
}

void Cal(int *arr) {
    for (int i=0;i<MAX_LEN;i++) {
        res+=arr[i];
    }
    printf("%d",res);
}

void Ran() {
    int min=1;
    int max=100;
    srand(time((NULL)));
    for (int i=0;i<MAX_LEN;i++) {
        arr[i]=rand()%(max-min+1);
    }
}
